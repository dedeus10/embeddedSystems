368 static inline uint8_t system_pinmux_pin_get_mux_position(
369  const uint8_t gpio_pin)
370 {
371  PortGroup *const port = system_pinmux_get_group_from_gpio_pin(gpio_pin);
372  uint32_t pin_index = (gpio_pin % 32);
373 
374  if (!(port->PINCFG[pin_index].reg & PORT_PINCFG_PMUXEN)) {
375  return SYSTEM_PINMUX_GPIO;
376  }
377 
378  uint32_t pmux_reg = port->PMUX[pin_index / 2].reg;
379 
380  if (pin_index & 1) {
381  return (pmux_reg & PORT_PMUX_PMUXO_Msk) >> PORT_PMUX_PMUXO_Pos;
382  }
383  else {
384  return (pmux_reg & PORT_PMUX_PMUXE_Msk) >> PORT_PMUX_PMUXE_Pos;
385  }
386 }
