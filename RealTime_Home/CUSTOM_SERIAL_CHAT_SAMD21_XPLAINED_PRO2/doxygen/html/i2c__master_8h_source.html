313 static inline enum status_code i2c_master_lock(
314  struct i2c_master_module *const module)
315 {
316  enum status_code status;
317 
318  system_interrupt_enter_critical_section();

320  if (module->locked) {

471 static inline void i2c_master_enable(
472  const struct i2c_master_module *const module)
473 {
474  /* Sanity check of arguments */
475  Assert(module);
476  Assert(module->hw);
477 
478  SercomI2cm *const i2c_module = &(module->hw->I2CM);
479 
480  /* Timeout counter used to force bus state */
481  uint32_t timeout_counter = 0;
482 
483  /* Wait for module to sync */
484  _i2c_master_wait_for_sync(module);
485 
486  /* Enable module */
487  i2c_module->CTRLA.reg |= SERCOM_I2CM_CTRLA_ENABLE;
488 
489 #if I2C_MASTER_CALLBACK_MODE == true
490  /* Enable module interrupts */
491  system_interrupt_enable(_sercom_get_interrupt_vector(module->hw));
492 #endif
493  /* Start timeout if bus state is unknown */
494  while (!(i2c_module->STATUS.reg & SERCOM_I2CM_STATUS_BUSSTATE(1))) {
495  timeout_counter++;
496  if(timeout_counter >= (module->unknown_bus_state_timeout)) {
497  /* Timeout, force bus state to idle */
498  i2c_module->STATUS.reg = SERCOM_I2CM_STATUS_BUSSTATE(1);
499  /* Workaround #1 */
500  return;
501  }
502  }
503 }

528
